\documentclass[
	a4paper, % Paper size, use 'a4paper' for A4 (default) or 'letterpaper' for US letter

%---------------------------------------------------------------------------------
%	SIGN CONTENT
%---------------------------------------------------------------------------------

\signtitle{Building Under Renovation} % The main sign title at the top

\signsubtitle{Facilities Moved to Building 42} % The bold subtitle under the main title, may be left empty (i.e. \signsubtitle{}) if not required

\descriptiontext{ % Paragraphs of description, explanation or instruction text, separated by empty lines for correct paragraph line spacing
	Renovation will continue through August if weather permits. In the event of delays, alternative office spaces and facilities will be provisioned. If you have any questions about how you will be impacted, please contact the facility manager.

	The shared staff kitchen and lunch room are now on level 10 of B42, shared with the Engineering Department.

	Showers are now on basement level 3 next to the elevators.

	Smoking is now allowed only on the balcony on level 14 or in Albert Park across the road.

	All facility tours are postponed until renovation is complete.
}

\additionaltext{Apologies for any inconvenience caused.} % Additional text at the bottom, may be left empty (i.e. \additionaltext{}) if not required
